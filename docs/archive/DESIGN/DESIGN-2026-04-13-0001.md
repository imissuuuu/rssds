**追加: config 読み込み関数群**

```bash
CONFIG_FILE=".claude/harness-config.json"

validate_config() {
  if [[ ! -f "$CONFIG_FILE" ]]; then
    echo "[wf] error: $CONFIG_FILE not found. Run: wf help" >&2
    exit 1
  fi
}

validate_flow() {
  validate_config
  local count
  count=$(jq -r --arg f "$1" '.workflows[$f].steps | length // 0' "$CONFIG_FILE")
  if [[ "$count" == "0" || "$count" == "null" ]]; then
    echo "[wf] unknown flow: $1 (available: $(jq -r '.workflows | keys | join(", ")' "$CONFIG_FILE"))" >&2
    exit 1
  fi
}

total_for() {
  jq -r --arg f "$1" '.workflows[$f].steps | length' "$CONFIG_FILE"
}

step_name() {
  local idx=$(($2 - 1))
  jq -r --arg f "$1" --argjson i "$idx" '.workflows[$f].steps[$i].name' "$CONFIG_FILE"
}

step_message() {
  local idx=$(($2 - 1))
  jq -r --arg f "$1" --argjson i "$idx" '.workflows[$f].steps[$i].message' "$CONFIG_FILE"
}
```

**cmd_status 変更 — message を含める:**

```bash
cmd_status() {
  if [[ ! -f "$STATE_FILE" ]]; then return 0; fi
  local flow task step total
  flow=$(read_field flow); task=$(read_field task)
  step=$(read_field step); total=$(read_field total)
  if [[ "$step" == "done" ]]; then
    echo "[wf] $flow/$task COMPLETED"; return 0
  fi
  validate_config
  local cur; cur=$(step_name "$flow" "$step")
  local msg; msg=$(step_message "$flow" "$step")
  if (( step < total )); then
    local nxt; nxt=$(step_name "$flow" "$((step + 1))")
    echo "[wf] $flow/$task step $step/$total: $cur ($msg) -> next: $nxt"
  else
    echo "[wf] $flow/$task step $step/$total: $cur ($msg) (final)"
  fi
}
```

**cmd_help 追加 + dispatch 更新:**

```bash
cmd_help() {
  echo "[wf] Workflow engine — see README.md for full documentation"
  echo ""
  echo "Commands:"
  echo "  wf start <flow> <task>  Start a workflow"
  echo "  wf status               Show current step"
  echo "  wf next                 Advance to next step"
  echo "  wf abort --force        Abort current workflow"
  echo "  wf help                 Show this help"
  echo ""
  if [[ -f "$CONFIG_FILE" ]]; then
    echo "Available flows: $(jq -r '.workflows | keys | join(", ")' "$CONFIG_FILE")"
  else
    echo "Config not found: $CONFIG_FILE"
  fi
}

case "${1:-status}" in
  start)      shift; cmd_start "$@" ;;
  status)     cmd_status ;;
  next)       cmd_next ;;
  abort)      shift; cmd_abort "$@" ;;
  check-stop) cmd_check_stop ;;
  help)       cmd_help ;;
  *) echo "usage: wf {start|status|next|abort|help}" >&2; exit 1 ;;
esac
```

install_config() {
  local src="$SRC/.claude/harness-config.json"
  local dst="$TARGET/.claude/harness-config.json"
  mkdir -p "$(dirname "$dst")"
  if [[ ! -f "$dst" ]]; then
    cp -a "$src" "$dst"
    info "harness-config.json: created"
  else
    info "harness-config.json: exists, skipped"
  fi
}
