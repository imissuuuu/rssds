# システム設計書 - Phase 6 ステータスバー + 設定画面
**RDD:** `RDD-2026-04-14-0001.md`
**作成日:** 2026-04-14

## 1. 全体方針

- 設定の読み書きは StreaMu の `ConfigManager` パターンを踏襲しつつ、rssds に既存の `nlohmann/json` を使用して実装
- ステータスバーは `ui.cpp` 内の static 変数（バッテリー・時刻キャッシュ）で管理し、`AppState` を汚染しない
- Settings 仮想エントリは `selectedFeed == feedConfigs.size()` として扱い、実フィードのインデックス範囲外であることを明示的にガード
- キー感度設定は ArticleView 遷移時の `computeRepeat()` 引数に直接適用（画面遷移ごとに切り替え済みの構造を維持）

### `source/settings.cpp`

```cpp
#include "settings.h"
#include <nlohmann/json.hpp>
#include <cstdio>
#include <sys/stat.h>

static const char* SETTINGS_PATH = "sdmc:/3ds/rssreader/settings.json";

bool settingsLoad(AppSettings& out) {
    out = AppSettings{};   // デフォルト値で初期化

    FILE* f = fopen(SETTINGS_PATH, "r");
    if (!f) return false;

    char buf[256];
    size_t len = fread(buf, 1, sizeof(buf) - 1, f);
    fclose(f);
    buf[len] = '\0';

    try {
        auto j = nlohmann::json::parse(buf);
        if (j.contains("scroll_repeat_delay_ms"))
            out.scrollRepeatDelayMs    = j["scroll_repeat_delay_ms"].get<int>();
        if (j.contains("scroll_repeat_interval_ms"))
            out.scrollRepeatIntervalMs = j["scroll_repeat_interval_ms"].get<int>();
    } catch (...) {
        out = AppSettings{};
        return false;
    }

    // バリデーション（範囲外はデフォルト値に戻す）
    const int VALID_DELAYS[]     = { 200, 300, 400, 500 };
    const int VALID_INTERVALS[]  = { 50, 80, 120, 160 };

    auto inList = [](int v, const int* arr, int n) {
        for (int i = 0; i < n; ++i) if (arr[i] == v) return true;
        return false;
    };
    if (!inList(out.scrollRepeatDelayMs,    VALID_DELAYS,    4)) out.scrollRepeatDelayMs    = 133;
    if (!inList(out.scrollRepeatIntervalMs, VALID_INTERVALS, 4)) out.scrollRepeatIntervalMs = 67;

    return true;
}

bool settingsSave(const AppSettings& cfg) {
    mkdir("sdmc:/3ds",            0777);
    mkdir("sdmc:/3ds/rssreader",  0777);

    FILE* f = fopen(SETTINGS_PATH, "w");
    if (!f) return false;

    nlohmann::json j;
    j["scroll_repeat_delay_ms"]    = cfg.scrollRepeatDelayMs;
    j["scroll_repeat_interval_ms"] = cfg.scrollRepeatIntervalMs;

    std::string s = j.dump(2);
    fwrite(s.c_str(), 1, s.size(), f);
    fclose(f);
    return true;
}
```

#### (C) `drawStatusBar()` ヘルパー（新規追加）

```cpp
static void drawStatusBar() {
    // バッテリー: 5秒ごとに更新
    u64 now = osGetTime();
    if (now - s_battLastMs > 5000) {
        PTMU_GetBatteryLevel(&s_battLevel);
        PTMU_GetBatteryChargeState(&s_battCharging);
        s_battLastMs = now;
    }

    // 時刻
    time_t t = time(nullptr);
    struct tm* tm_info = localtime(&t);
    char timeBuf[8];
    snprintf(timeBuf, sizeof(timeBuf), "%02d:%02d",
             tm_info->tm_hour, tm_info->tm_min);

    // バッテリー文字列
    char battBuf[16];
    int pct = (int)s_battLevel * 20;
    if (s_battCharging)
        snprintf(battBuf, sizeof(battBuf), "%d%%+", pct);
    else
        snprintf(battBuf, sizeof(battBuf), "%d%%",  pct);

    // 描画（背景帯 + テキスト）
    C2D_DrawRectSolid(0, 0, 0.5f, TOP_W, STATUSBAR_H, CLR_PANEL);
    drawText(timeBuf, TEXT_MARGIN_X, 1.0f, 0.5f, 0.42f, 0.42f, CLR_HINT);

    float battW = measureStr(battBuf, 0.42f);
    drawText(battBuf, TOP_W - battW - TEXT_MARGIN_X, 1.0f, 0.5f,
             0.42f, 0.42f, CLR_HINT);
}
```

#### (F) `drawSettings()` 新規追加

```cpp
static const int SCROLL_DELAY_OPTIONS[]    = { 200, 300, 400, 500 };
static const int SCROLL_INTERVAL_OPTIONS[] = { 50, 80, 120, 160 };
static const int NUM_DELAY_OPTS    = 4;
static const int NUM_INTERVAL_OPTS = 4;

static int findIndex(const int* arr, int n, int val) {
    for (int i = 0; i < n; ++i) if (arr[i] == val) return i;
    return 0;
}

static void drawSettings(const AppState& state) {
    // 上画面: タイトル
    C2D_TargetClear(topTarget, CLR_BG);
    C2D_SceneBegin(topTarget);
    drawStatusBar();
    drawStyledText("Settings", TEXT_MARGIN_X, TOP_CONTENT_Y, 0.5f, TextStyle::Heading);

    // 下画面: 設定項目
    C2D_TargetClear(botTarget, CLR_PANEL);
    C2D_SceneBegin(botTarget);

    // items[0]=Delay, items[1]=Interval, items[2]=Save
    struct Item { const char* label; int value; bool isAction; };
    Item items[3] = {
        { "Scroll Delay",    state.settings.scrollRepeatDelayMs,    false },
        { "Scroll Interval", state.settings.scrollRepeatIntervalMs, false },
        { "Save",            0,                                      true  },
    };

    for (int i = 0; i < 3; ++i) {
        float y = TEXT_MARGIN_Y + (float)i * LINE_HEIGHT;
        if (i == state.settingsSelectedItem) {
            C2D_DrawRectSolid(0, y - 1.0f, 0.0f, BOT_W, LINE_HEIGHT + 1.0f, CLR_SEL_BG);
        }
        char buf[64];
        if (items[i].isAction)
            snprintf(buf, sizeof(buf), "%s", items[i].label);
        else
            snprintf(buf, sizeof(buf), "%-18s < %d ms >", items[i].label, items[i].value);
        drawText(buf, TEXT_MARGIN_X, y, 0.5f, TEXT_SCALE, TEXT_SCALE, CLR_TEXT);
    }

    drawText("Up/Down:select  L/R:change  A:save  B:back",
             TEXT_MARGIN_X, BOT_H - 16.0f, 0.5f, 0.42f, 0.42f, CLR_HINT);
}
```

**Settings ケース追加:**
```cpp
case Screen::Settings: {
    if ((kRepeat & KEY_UP)   && state.settingsSelectedItem > 0)
        --state.settingsSelectedItem;
    if ((kRepeat & KEY_DOWN) && state.settingsSelectedItem < 2)
        ++state.settingsSelectedItem;

    // 値変更（Save 行では無効）
    if (state.settingsSelectedItem == 0) {
        // Scroll Delay
        int idx = findIndex(SCROLL_DELAY_OPTIONS, NUM_DELAY_OPTS,
                            state.settings.scrollRepeatDelayMs);
        if ((kRepeat & KEY_RIGHT) && idx < NUM_DELAY_OPTS - 1)
            state.settings.scrollRepeatDelayMs = SCROLL_DELAY_OPTIONS[idx + 1];
        if ((kRepeat & KEY_LEFT)  && idx > 0)
            state.settings.scrollRepeatDelayMs = SCROLL_DELAY_OPTIONS[idx - 1];
    } else if (state.settingsSelectedItem == 1) {
        // Scroll Interval
        int idx = findIndex(SCROLL_INTERVAL_OPTIONS, NUM_INTERVAL_OPTS,
                            state.settings.scrollRepeatIntervalMs);
        if ((kRepeat & KEY_RIGHT) && idx < NUM_INTERVAL_OPTS - 1)
            state.settings.scrollRepeatIntervalMs = SCROLL_INTERVAL_OPTIONS[idx + 1];
        if ((kRepeat & KEY_LEFT)  && idx > 0)
            state.settings.scrollRepeatIntervalMs = SCROLL_INTERVAL_OPTIONS[idx - 1];
    }

    // Save 行で A → 保存して戻る
    if ((kDown & KEY_A) && state.settingsSelectedItem == 2) {
        settingsSave(state.settings);
        state.currentScreen = Screen::FeedList;
        kDown &= ~KEY_A;
    }
    // B → 保存せずに戻る
    if (kDown & KEY_B) {
        state.currentScreen = Screen::FeedList;
        kDown &= ~KEY_B;
    }
    break;
}
```

## 4. 各機能の詳細設計

### FR-1: ステータスバー

```
uiDraw()
  └─ drawXxx()
       ├─ C2D_TargetClear(topTarget, CLR_BG)
       ├─ C2D_SceneBegin(topTarget)
       ├─ drawStatusBar()          ← y=0..14 に描画
       └─ 既存コンテンツ（y=TOP_CONTENT_Y=20 以降）
```

バッテリー取得フロー:
```
drawStatusBar() 呼び出しごとに:
  if osGetTime() - s_battLastMs > 5000:
    PTMU_GetBatteryLevel(&s_battLevel)       → 0-5
    PTMU_GetBatteryChargeState(&s_battCharging)
    s_battLastMs = now
  pct = s_battLevel * 20  (0%〜100%)
```

### FR-2/3: Settings エントリ + 設定画面遷移

```
FeedList
  selectedFeed == feedCount (Settings 行) かつ A押下
    → Screen::Settings, settingsSelectedItem = 0

Settings
  Up/Down → settingsSelectedItem (0=Delay, 1=Interval)
  Left/Right → 選択肢を1ステップ変更（即時 state 反映、ディスク未書込）
  A (Save 行選択時) → settingsSave() → Screen::FeedList
  B → 保存せず Screen::FeedList へ戻る（変更は state に残るが次回起動時には元の値）
```

### FR-4: キー感度の反映タイミング

設定値は `state.settings` に常駐。`main.cpp` のメインループが毎フレーム参照するため、**次フレームから即時反映**される（明示的な「適用」処理は不要）。
